# Weeks and Days
for w in {1..4}; do
  mkdir -p Week$w
  d1=$(( (w-1)*7 + 1 ))
  d2=$(( w*7 ))
  for d in $(seq -w $d1 $d2); do
    mkdir -p Week$w/Day$d
  done
done
